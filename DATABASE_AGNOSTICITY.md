## 1. Core Philosophy: Repository Pattern

### The Contract: One Interface, Many Implementations

backbone enforces the **Repository Pattern** to ensure your application layer never knows which database you're using.

```
┌──────────────────────────────┐
│   Application Layer          │
│   (Business Logic)           │
│   - Commands, Queries        │
│   - Speaks interface         │
└──────────────────┬───────────┘
                   │
        ┌──────────▼──────────┐
        │ Repository Interface│
        │ (Abstract)          │
        └──────────┬──────────┘
                   │
    ┌──────────────┼──────────────┬──────────────┐
    │              │              │              │
┌───▼──┐   ┌──────▼────┐  ┌──────▼────┐  ┌──────▼────┐
│ SQL  │   │  NoSQL    │  │ Key-Value │  │  Cloud    │
│ Impl │   │  Impl     │  │  Impl     │  │  Impl     │
└──────┘   └───────────┘  └───────────┘  └───────────┘
```

**The application layer calls:**
```go
products, _ := repo.FindByCriteria(ctx, criteria)
```

**No matter what's behind the repo:**
- PostgreSQL implementation
- MongoDB implementation
- DynamoDB implementation
- Firestore implementation
- In-memory test implementation

**All return the same type, in the same order.**

## 2. Database-Agnostic Design

### The Repository Interface (Universal)

Both Go and Python use the same interface:

**Go:**
```go
package domain

import (
    "context"
    "your-app/domain/specifications"
)

type ProductRepository interface {
    FindByID(ctx context.Context, id string) (*Product, error)
    FindByName(ctx context.Context, name string) (*Product, error)
    FindByCriteria(ctx context.Context, c *specifications.Criteria) ([]*Product, error)
    Count(ctx context.Context, c *specifications.Criteria) (int, error)
    Create(ctx context.Context, p *Product) error
    Update(ctx context.Context, p *Product) error
    Delete(ctx context.Context, id string) error
}
```

**Python:**
```python
from abc import ABC, abstractmethod
from typing import List, Optional
from your_app.domain.specifications import Criteria

class ProductRepository(ABC):
    @abstractmethod
    def find_by_id(self, product_id: str) -> Optional[Product]:
        pass

    @abstractmethod
    def find_by_name(self, name: str) -> Optional[Product]:
        pass

    @abstractmethod
    def find_by_criteria(self, criteria: Criteria) -> List[Product]:
        pass

    @abstractmethod
    def count(self, criteria: Criteria) -> int:
        pass

    @abstractmethod
    def create(self, product: Product) -> None:
        pass

    @abstractmethod
    def update(self, product: Product) -> None:
        pass

    @abstractmethod
    def delete(self, product_id: str) -> None:
        pass
```

**Key insight:** The interface knows nothing about SQL, MongoDB, DynamoDB, etc. Only the implementations do.

### Go: PostgreSQL with SQLC

```go
package infrastructure

import (
    "context"
    "database/sql"
    "fmt"

type PostgresProductRepository struct {
    db      *sql.DB

func (r *PostgresProductRepository) FindByID(ctx context.Context, id string) (*domain.Product, error) {
    row, err := r.queries.GetProductByID(ctx, id)
    if err != nil {
        if err == sql.ErrNoRows {
            return nil, nil
        }
        return nil, fmt.Errorf("database error: %w", err)
    }

    return &domain.Product{
        ID:       row.ID,
        Name:     row.Name,
        Price:    row.Price,
        Status:   row.Status,
        Category: row.Category,
    }, nil
}

func (r *PostgresProductRepository) FindByName(ctx context.Context, name string) (*domain.Product, error) {
    row, err := r.queries.GetProductByName(ctx, name)
    if err != nil {
        if err == sql.ErrNoRows {
            return nil, nil
        }
        return nil, fmt.Errorf("database error: %w", err)
    }

    return &domain.Product{
        ID:       row.ID,
        Name:     row.Name,
        Price:    row.Price,
        Status:   row.Status,
        Category: row.Category,
    }, nil
}

func (r *PostgresProductRepository) FindByCriteria(ctx context.Context, c *specifications.Criteria) ([]*domain.Product, error) {
    // Build SQL query from Criteria
    query := "SELECT id, name, price, status, category FROM products WHERE 1=1"
    args := []interface{}{}
    argIndex := 1

    // Apply filters from specification
    if c.Specification != nil {
        filterSQL, filterArgs := r.buildFilterSQL(c.Specification)
        query += " AND " + filterSQL
        args = append(args, filterArgs...)
        argIndex += len(filterArgs)
    }

    // Apply sorting
    if c.SortField != "" {
        direction := "ASC"
        if c.SortDirection == "desc" {
            direction = "DESC"
        }
        query += fmt.Sprintf(" ORDER BY %s %s", c.SortField, direction)
    }

    // Apply pagination
    offset := (c.Page - 1) * c.PageSize
    query += fmt.Sprintf(" LIMIT $%d OFFSET $%d", argIndex, argIndex+1)
    args = append(args, c.PageSize, offset)

    // Execute query
    rows, err := r.db.QueryContext(ctx, query, args...)
    if err != nil {
        return nil, fmt.Errorf("query error: %w", err)
    }
    defer rows.Close()

    var products []*domain.Product
    for rows.Next() {
        var p domain.Product
        err := rows.Scan(&p.ID, &p.Name, &p.Price, &p.Status, &p.Category)
        if err != nil {
            return nil, fmt.Errorf("scan error: %w", err)
        }
        products = append(products, &p)
    }

    return products, rows.Err()
}

func (r *PostgresProductRepository) Count(ctx context.Context, c *specifications.Criteria) (int, error) {
    query := "SELECT COUNT(*) FROM products WHERE 1=1"
    args := []interface{}{}

    if c.Specification != nil {
        filterSQL, filterArgs := r.buildFilterSQL(c.Specification)
        query += " AND " + filterSQL
        args = append(args, filterArgs...)
    }

    var count int
    err := r.db.QueryRowContext(ctx, query, args...).Scan(&count)
    if err != nil {
        return 0, fmt.Errorf("count error: %w", err)
    }

    return count, nil
}

func (r *PostgresProductRepository) Create(ctx context.Context, p *domain.Product) error {
    _, err := r.queries.CreateProduct(ctx, queries.CreateProductParams{
        ID:       p.ID,
        Name:     p.Name,
        Price:    p.Price,
        Status:   p.Status,
        Category: p.Category,

    if err != nil {

    }

}

    err := r.queries.UpdateProduct(ctx, queries.UpdateProductParams{

    if err != nil {
```

### Python: PostgreSQL with SQLAlchemy

```python
from typing import List, Optional
from sqlalchemy import create_engine, select, and_
from sqlalchemy.orm import Session
from your_app.domain import Product
from your_app.domain.specifications import Criteria

class PostgresProductRepository:
    def __init__(self, database_url: str):
        self.engine = create_engine(database_url)

    def find_by_id(self, product_id: str) -> Optional[Product]:
        with Session(self.engine) as session:
            stmt = select(ProductModel).where(ProductModel.id == product_id)
            result = session.execute(stmt).scalar_one_or_none()
            return self._to_domain(result) if result else None

    def find_by_name(self, name: str) -> Optional[Product]:
        with Session(self.engine) as session:
            stmt = select(ProductModel).where(ProductModel.name == name)
            result = session.execute(stmt).scalar_one_or_none()
            return self._to_domain(result) if result else None

    def find_by_criteria(self, criteria: Criteria) -> List[Product]:
        with Session(self.engine) as session:
            query = select(ProductModel)

            # Apply filters
            if criteria.specification:
                filters = self._build_filters(criteria.specification)
                query = query.where(and_(*filters))

            # Apply sorting
            if criteria.sort_field:
                order_column = getattr(ProductModel, criteria.sort_field)
                if criteria.sort_direction.lower() == "desc":
                    query = query.order_by(order_column.desc())
                else:
                    query = query.order_by(order_column.asc())

            # Apply pagination
            offset = (criteria.page - 1) * criteria.page_size
            query = query.limit(criteria.page_size).offset(offset)

            # Execute
            results = session.execute(query).scalars().all()
            return [self._to_domain(r) for r in results]

    def count(self, criteria: Criteria) -> int:
        with Session(self.engine) as session:
            query = select(func.count()).select_from(ProductModel)

            if criteria.specification:
                filters = self._build_filters(criteria.specification)
                query = query.where(and_(*filters))

            return session.execute(query).scalar()

    def create(self, product: Product) -> None:
        with Session(self.engine) as session:
            model = ProductModel(
                id=product.id,
                name=product.name,
                price=product.price,
                status=product.status,
                category=product.category,
            )
            session.add(model)
            session.commit()

    def update(self, product: Product) -> None:
        with Session(self.engine) as session:
            model = session.execute(
                select(ProductModel).where(ProductModel.id == product.id)
            ).scalar_one()

            model.name = product.name
            model.price = product.price
            model.status = product.status
            model.category = product.category

            session.commit()

    def delete(self, product_id: str) -> None:
        with Session(self.engine) as session:
            model = session.execute(
                select(ProductModel).where(ProductModel.id == product_id)
            ).scalar_one()
            session.delete(model)
            session.commit()

    @staticmethod
    def _to_domain(model) -> Product:
        return Product(
            id=model.id,
            name=model.name,
            price=model.price,
            status=model.status,
            category=model.category,
        )

    @staticmethod
    def _build_filters(specification):
        # Convert specification to SQLAlchemy filters
        return [ProductModel.status == "active"]  # Simplified
```

### Go: MongoDB Native Driver

```go
package infrastructure

import (
    "context"
    "fmt"
    "your-app/domain"
    "your-app/domain/specifications"
    "go.mongodb.org/mongo-driver/bson"
    "go.mongodb.org/mongo-driver/mongo"
    "go.mongodb.org/mongo-driver/mongo/options"
)

type MongoProductRepository struct {
    collection *mongo.Collection
}

func NewMongoProductRepository(client *mongo.Client) *MongoProductRepository {
    return &MongoProductRepository{
        collection: client.Database("products_db").Collection("products"),
    }
}

func (r *MongoProductRepository) FindByID(ctx context.Context, id string) (*domain.Product, error) {
    var result ProductDoc
    err := r.collection.FindOne(ctx, bson.M{"_id": id}).Decode(&result)
    if err != nil {
        if err == mongo.ErrNoDocuments {
            return nil, nil
        }
        return nil, fmt.Errorf("mongo error: %w", err)
    }

    return &domain.Product{
        ID:       result.ID,
        Name:     result.Name,
        Price:    result.Price,
        Status:   result.Status,
        Category: result.Category,
    }, nil
}

func (r *MongoProductRepository) FindByCriteria(ctx context.Context, c *specifications.Criteria) ([]*domain.Product, error) {
    // Build MongoDB filter from Criteria
    filter := bson.M{}

    if c.Specification != nil {
        filter = r.buildMongoFilter(c.Specification)
    }

    // Build options (sort, pagination)
    opts := options.Find()

    if c.SortField != "" {
        direction := int32(1)
        if c.SortDirection == "desc" {
            direction = -1
        }
        opts.SetSort(bson.M{c.SortField: direction})
    }

    skip := int64((c.Page - 1) * c.PageSize)
    opts.SetSkip(skip).SetLimit(int64(c.PageSize))

    // Query
    cursor, err := r.collection.Find(ctx, filter, opts)
    if err != nil {
        return nil, fmt.Errorf("find error: %w", err)
    }
    defer cursor.Close(ctx)

    var products []*domain.Product
    if err = cursor.All(ctx, &products); err != nil {
        return nil, fmt.Errorf("decode error: %w", err)
    }

    return products, nil
}

func (r *MongoProductRepository) Count(ctx context.Context, c *specifications.Criteria) (int, error) {
    filter := bson.M{}
    if c.Specification != nil {
        filter = r.buildMongoFilter(c.Specification)
    }

    count, err := r.collection.CountDocuments(ctx, filter)
    if err != nil {
        return 0, fmt.Errorf("count error: %w", err)
    }

    return int(count), nil
}

func (r *MongoProductRepository) Create(ctx context.Context, p *domain.Product) error {

        "_id":      p.ID,
        "name":     p.Name,
        "price":    p.Price,

        "category": p.Category,

        return fmt.Errorf("insert error: %w", err)

    return nil
}

func (r *MongoProductRepository) Update(ctx context.Context, p *domain.Product) error {

        bson.M{"_id": p.ID},
        bson.M{"$set": bson.M{
            "name":     p.Name,
            "price":    p.Price,
            "status":   p.Status,
            "category": p.Category,
        }},
    )
    if err != nil {
        return fmt.Errorf("update error: %w", err)
    }
    return nil
}

func (r *MongoProductRepository) Delete(ctx context.Context, id string) error {
    _, err := r.collection.DeleteOne(ctx, bson.M{"_id": id})
    if err != nil {
        return fmt.Errorf("delete error: %w", err)
    }
    return nil
}
```

### Python: MongoDB with PyMongo

```python
from typing import List, Optional
from pymongo import MongoClient
from pymongo.errors import DuplicateKeyError
from your_app.domain import Product
from your_app.domain.specifications import Criteria

class MongoProductRepository:
    def __init__(self, mongo_uri: str):
        self.client = MongoClient(mongo_uri)
        self.db = self.client["products_db"]
        self.collection = self.db["products"]

    def find_by_id(self, product_id: str) -> Optional[Product]:
        doc = self.collection.find_one({"_id": product_id})
        return self._to_domain(doc) if doc else None

    def find_by_name(self, name: str) -> Optional[Product]:
        doc = self.collection.find_one({"name": name})
        return self._to_domain(doc) if doc else None

    def find_by_criteria(self, criteria: Criteria) -> List[Product]:
        # Build MongoDB filter
        mongo_filter = self._build_filter(criteria.specification)

        # Build sort
        sort = []
        if criteria.sort_field:
            direction = -1 if criteria.sort_direction.lower() == "desc" else 1
            sort = [(criteria.sort_field, direction)]

        # Execute query with pagination
        skip = (criteria.page - 1) * criteria.page_size
        cursor = self.collection.find(mongo_filter)
        if sort:
            cursor = cursor.sort(sort)
        cursor = cursor.skip(skip).limit(criteria.page_size)

        return [self._to_domain(doc) for doc in cursor]

    def count(self, criteria: Criteria) -> int:
        mongo_filter = self._build_filter(criteria.specification)
        return self.collection.count_documents(mongo_filter)

    def create(self, product: Product) -> None:
        try:
            self.collection.insert_one({
                "_id": product.id,
                "name": product.name,
                "price": product.price,
                "status": product.status,
                "category": product.category,
            })
        except DuplicateKeyError:
            raise Exception(f"Product {product.id} already exists")

    def update(self, product: Product) -> None:
        self.collection.update_one(
            {"_id": product.id},
            {"$set": {
                "name": product.name,
                "price": product.price,
                "status": product.status,
                "category": product.category,
            }}
        )

    def delete(self, product_id: str) -> None:
        self.collection.delete_one({"_id": product_id})

    @staticmethod
    def _to_domain(doc) -> Product:
        return Product(
            id=doc["_id"],
            name=doc["name"],
            price=doc["price"],
            status=doc["status"],
            category=doc["category"],
        )

    @staticmethod
    def _build_filter(specification):
        # Convert specification to MongoDB filter
        return {"status": "active"}  # Simplified
```

```go
package infrastructure

import (
    "context"
    "fmt"
    "your-app/domain"
    "your-app/domain/specifications"
    "github.com/aws/aws-sdk-go-v2/aws"
    "github.com/aws/aws-sdk-go-v2/feature/dynamodb/attributevalue"
    "github.com/aws/aws-sdk-go-v2/service/dynamodb"
    "github.com/aws/aws-sdk-go-v2/service/dynamodb/types"
)

type DynamoProductRepository struct {
    client *dynamodb.Client
}

func NewDynamoProductRepository(client *dynamodb.Client) *DynamoProductRepository {
    return &DynamoProductRepository{client: client}
}

func (r *DynamoProductRepository) FindByID(ctx context.Context, id string) (*domain.Product, error) {
    output, err := r.client.GetItem(ctx, &dynamodb.GetItemInput{
        TableName: aws.String("Products"),
        Key: map[string]types.AttributeValue{
            "id": &types.AttributeValueMemberS{Value: id},
        },
    })
    if err != nil {
        return nil, fmt.Errorf("dynamodb error: %w", err)
    }

    if output.Item == nil {
        return nil, nil
    }

    var p domain.Product
    err = attributevalue.UnmarshalMap(output.Item, &p)
    if err != nil {
        return nil, fmt.Errorf("unmarshal error: %w", err)
    }

    return &p, nil
}

func (r *DynamoProductRepository) FindByCriteria(ctx context.Context, c *specifications.Criteria) ([]*domain.Product, error) {
    // DynamoDB limitations: can only filter by partition key directly or use Scan with filter
    // For complex queries, use Scan with FilterExpression

    // Build filter expression from criteria
    filterExpr := "#status = :status"
    exprAttrNames := map[string]string{"#status": "status"}
    exprAttrValues := map[string]types.AttributeValue{
        ":status": &types.AttributeValueMemberS{Value: "active"},
    }

    output, err := r.client.Scan(ctx, &dynamodb.ScanInput{
        TableName:                 aws.String("Products"),
        FilterExpression:          aws.String(filterExpr),
        ExpressionAttributeNames:  exprAttrNames,
        ExpressionAttributeValues: exprAttrValues,
        Limit:                     aws.Int32(int32(c.PageSize)),
    })
    if err != nil {
        return nil, fmt.Errorf("scan error: %w", err)
    }

    var products []*domain.Product
    err = attributevalue.UnmarshalListOfMaps(output.Items, &products)
    if err != nil {
        return nil, fmt.Errorf("unmarshal error: %w", err)
    }

    return products, nil
}

func (r *DynamoProductRepository) Count(ctx context.Context, c *specifications.Criteria) (int, error) {
    output, err := r.client.Scan(ctx, &dynamodb.ScanInput{
        TableName:      aws.String("Products"),
        Select:         types.SelectCount,
        ConsistentRead: aws.Bool(true),
    })
    if err != nil {
        return 0, fmt.Errorf("count error: %w", err)
    }

    return int(*output.Count), nil
}

func (r *DynamoProductRepository) Create(ctx context.Context, p *domain.Product) error {
    av, err := attributevalue.MarshalMap(p)
    if err != nil {
        return fmt.Errorf("marshal error: %w", err)
    }

    _, err = r.client.PutItem(ctx, &dynamodb.PutItemInput{
        TableName: aws.String("Products"),
        Item:      av,
    })
    if err != nil {
        return fmt.Errorf("put error: %w", err)
    }

    return nil
}

func (r *DynamoProductRepository) Update(ctx context.Context, p *domain.Product) error {
    av, err := attributevalue.MarshalMap(p)
    if err != nil {
        return fmt.Errorf("marshal error: %w", err)
    }

    _, err = r.client.PutItem(ctx, &dynamodb.PutItemInput{
        TableName: aws.String("Products"),
        Item:      av,
    })
    if err != nil {
        return fmt.Errorf("update error: %w", err)
    }

    return nil
}

func (r *DynamoProductRepository) Delete(ctx context.Context, id string) error {
    _, err := r.client.DeleteItem(ctx, &dynamodb.DeleteItemInput{
        TableName: aws.String("Products"),
        Key: map[string]types.AttributeValue{
            "id": &types.AttributeValueMemberS{Value: id},
        },
    })
    if err != nil {
        return fmt.Errorf("delete error: %w", err)
    }

    return nil
}
```

### Python: AWS DynamoDB with boto3

```python
from typing import List, Optional
import boto3
from your_app.domain import Product
from your_app.domain.specifications import Criteria

class DynamoProductRepository:
    def __init__(self, region: str = "us-east-1"):
        self.dynamodb = boto3.resource("dynamodb", region_name=region)
        self.table = self.dynamodb.Table("Products")

    def find_by_id(self, product_id: str) -> Optional[Product]:
        response = self.table.get_item(Key={"id": product_id})
        if "Item" not in response:
            return None
        return self._to_domain(response["Item"])

    def find_by_name(self, name: str) -> Optional[Product]:
        # DynamoDB: Scan with filter (inefficient for large tables)
        response = self.table.scan(
            FilterExpression="attribute_exists(#n) AND #n = :name",
            ExpressionAttributeNames={"#n": "name"},
            ExpressionAttributeValues={":name": name}
        )
        if response["Items"]:
            return self._to_domain(response["Items"][0])
        return None

    def find_by_criteria(self, criteria: Criteria) -> List[Product]:
        # Build filter expression
        filter_expr = "#status = :status"
        attr_names = {"#status": "status"}
        attr_values = {":status": "active"}

        response = self.table.scan(
            FilterExpression=filter_expr,
            ExpressionAttributeNames=attr_names,
            ExpressionAttributeValues=attr_values,
            Limit=criteria.page_size
        )

        return [self._to_domain(item) for item in response["Items"]]

    def count(self, criteria: Criteria) -> int:
        response = self.table.scan(Select="COUNT")
        return response["Count"]

    def create(self, product: Product) -> None:
        self.table.put_item(Item={
            "id": product.id,
            "name": product.name,
            "price": product.price,
            "status": product.status,
            "category": product.category,
        })

    def update(self, product: Product) -> None:
        self.table.update_item(
            Key={"id": product.id},
            UpdateExpression="SET #n = :n, #p = :p, #s = :s, #c = :c",
            ExpressionAttributeNames={
                "#n": "name",
                "#p": "price",

            ExpressionAttributeValues={

            }
        )
```

## 6. Implementation: Firestore (Cloud Document)

### Python: Google Cloud Firestore

```python
from typing import List, Optional
from google.cloud import firestore
from your_app.domain import Product
from your_app.domain.specifications import Criteria

class FirestoreProductRepository:
    def __init__(self, project_id: str = None):
        self.db = firestore.Client(project=project_id)
        self.collection = self.db.collection("products")

    def find_by_id(self, product_id: str) -> Optional[Product]:
        doc = self.collection.document(product_id).get()
        if not doc.exists:
            return None
        return self._to_domain(doc.to_dict())

    def find_by_name(self, name: str) -> Optional[Product]:
        docs = self.collection.where("name", "==", name).limit(1).stream()
        for doc in docs:
            return self._to_domain(doc.to_dict())
        return None

    def find_by_criteria(self, criteria: Criteria) -> List[Product]:
        # Build Firestore query
        query = self.collection

        # Apply filters
        if criteria.specification:
            query = query.where("status", "==", "active")  # Simplified

        # Apply sorting
        if criteria.sort_field:
            direction = firestore.Query.DESCENDING if criteria.sort_direction.lower() == "desc" else firestore.Query.ASCENDING
            query = query.order_by(criteria.sort_field, direction=direction)

        # Apply pagination
        offset = (criteria.page - 1) * criteria.page_size
        query = query.offset(offset).limit(criteria.page_size)

        # Execute
        docs = query.stream()
        return [self._to_domain(doc.to_dict()) for doc in docs]

    def count(self, criteria: Criteria) -> int:
        query = self.collection
        if criteria.specification:
            query = query.where("status", "==", "active")
        
        return query.count().get()[0][0].value

    def create(self, product: Product) -> None:
        self.collection.document(product.id).set({
            "name": product.name,
            "price": product.price,
            "status": product.status,
            "category": product.category,
        })

    def update(self, product: Product) -> None:
        self.collection.document(product.id).update({
            "name": product.name,
            "price": product.price,
            "status": product.status,
            "category": product.category,
        })

    def delete(self, product_id: str) -> None:
        self.collection.document(product_id).delete()

    @staticmethod
    def _to_domain(data) -> Product:
        return Product(
            id=data["id"],  # Assumes id is stored in document
            name=data["name"],
            price=data["price"],
            status=data["status"],
            category=data["category"],
        )
```

```go
// config/repositories.go
package config

import (
    "your-app/domain"
    "your-app/infrastructure"

func ProvideProductRepository(cfg *Config) domain.ProductRepository {
    switch cfg.Database.Type {
    case "postgres":

    case "mongodb":
        return infrastructure.NewMongoProductRepository(cfg.Database.MongoClient)
    case "dynamodb":
        return infrastructure.NewDynamoProductRepository(cfg.Database.DynamoClient)
    case "firestore":

    }
}
```

### Data Migration Pattern

When migrating from PostgreSQL to MongoDB:

```python
# scripts/migrate_to_mongo.py
from your_app.infrastructure import PostgresProductRepository, MongoProductRepository
from your_app.domain.specifications import Criteria

def migrate_all_products(page_size=1000):
    postgres_repo = PostgresProductRepository(os.getenv("PG_URL"))
    mongo_repo = MongoProductRepository(os.getenv("MONGO_URI"))

    page = 1
    while True:
        # Read from PostgreSQL
        criteria = Criteria(page=page, page_size=page_size)
        products = postgres_repo.find_by_criteria(criteria)

        if not products:
            break

        # Write to MongoDB
        for product in products:
            mongo_repo.create(product)

        page += 1

    print(f"Migration complete: {(page - 1) * page_size} products migrated")
```

## 8. Performance Considerations

### Database Selection Guide

| Database | Use Case | Strengths | Limitations |
|----------|----------|-----------|-------------|
| **PostgreSQL** | OLTP, complex queries | ACID, powerful queries | Scaling writes is hard |
| **MongoDB** | Flexible schema, documents | Flexible schema, easy sharding | Complex joins, transactions |
| **DynamoDB** | High-throughput, serverless | Auto-scaling, serverless, managed | Limited querying, expensive |
| **Firestore** | Real-time, mobile, web | Real-time sync, serverless | Limited query operators, cost |
| **ElasticSearch** | Full-text search | Fast search, analytics | Write-heavy, eventual consistency |

### Optimization Strategies

**PostgreSQL:**
```sql
-- Add indexes for common filters
CREATE INDEX idx_status ON products(status);
CREATE INDEX idx_category ON products(category);
CREATE INDEX idx_created_at ON products(created_at DESC);

-- Use EXPLAIN ANALYZE to optimize queries
EXPLAIN ANALYZE
SELECT * FROM products WHERE category = 'Electronics' AND price BETWEEN 500 AND 2000
ORDER BY price DESC LIMIT 10;
```

**MongoDB:**
```python
# Add indexes for common filters
db.products.create_index([("status", 1)])
db.products.create_index([("category", 1)])
db.products.create_index([("created_at", -1)])

# Use compound indexes for common filter combinations
db.products.create_index([("status", 1), ("category", 1), ("price", 1)])
```

**DynamoDB:**
```python
# Design partition key strategically
# Primary key: id (partition key)
# GSI 1: status (partition) + created_at (sort)
# GSI 2: category (partition) + price (sort)

# This enables efficient queries on status + category + price
```

### Pagination Performance

**PostgreSQL/MongoDB:** Offset-based is simple but slow for large offsets
```sql
-- SLOW for page 1000
SELECT * FROM products LIMIT 10 OFFSET 10000;

-- BETTER: Cursor-based pagination
SELECT * FROM products WHERE id > :last_id ORDER BY id LIMIT 10;
```

**DynamoDB:** Use LastEvaluatedKey for efficient pagination
```python
response = table.scan(Limit=10)
last_key = response.get('LastEvaluatedKey')  # Use for next page

# Next page:
response = table.scan(Limit=10, ExclusiveStartKey=last_key)
```
